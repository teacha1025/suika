746 template<concepts::numbers T>
747 double determinant(const matrix<T>& mt) {
748 if (mt.size().x != mt.size().y) {
749 throw;
750 }
751 uint n = mt.size().x;
752 if (n == 1) {
753 return mt.at(0, 0);
754 }
755 else if (n == 2) {
756 return mt.at(0, 0) * mt.at(1, 1) - mt.at(0, 1) * mt.at(1, 0);
757 }
758 else if (n == 3) {
759 return mt.at(0, 0) * mt.at(1, 1) * mt.at(2, 2) + mt.at(0, 1) * mt.at(1, 2) * mt.at(2, 0) + mt.at(0, 2) * mt.at(1, 0) * mt.at(2, 1)
760 - mt.at(0, 0) * mt.at(1, 2) * mt.at(2, 1) - mt.at(0, 1) * mt.at(1, 0) * mt.at(2, 2) - mt.at(0, 2) * mt.at(1, 1) * mt.at(2, 0);
761 }
762 else {
763 double det = 1, buf;
764 matrix<double> tmp = mt;
765
766 for (uint i = 0; i < n; i++) {
767 for (uint j = 0; j < n; j++) {
768 if (i < j) {
769 if (!tmp.at(i, i)) {
770 double max = 0;
771 int max_i = i;
772 for (uint k = i + 1; k < n; k++) {
773 if (max < tmp.at(k, i)) {
774 max = tmp.at(k, i);
775 max_i = k;
776 }
777 }
778 if (max == 0) {
779 throw;
780 }
781 for (uint k = 0; k < n; k++) {
782 std::swap(tmp.at(i, k), tmp.at(max_i, k));
783 }
784 det *= -1;
785 }
786 buf = tmp.at(j, i) / tmp.at(i, i);
787 for (uint k = 0; k < n; k++) {
788 tmp.at(j, k) -= tmp.at(i, k) * buf;
789 }
790 }
791 }
792 }
793
794 for (uint i = 0; i < n; i++) {
795 det *= tmp.at(i, i);
796 }
797
798 return det;
799 }

807 template<concepts::numbers T>
808 matrix<T> inverse(const matrix<T>& mt) {
809 if (mt.size().x != mt.size().y) {
810 throw;
811 }
812 uint n = mt.size().x;
813
814
815 if (n == 1) {
816 if (!mt.at(0, 0)) {
817 throw;
818 }
819 return 1.0 / mt;
820 }
821 else if (n == 2) {
822 matrix<T> ret(2, 2, { mt.at(1, 1),-mt.at(0, 1),-mt.at(1, 0),mt.at(0, 0) });
823 if (auto t = mt.at(0, 0) * mt.at(1, 1) - mt.at(0, 1) * mt.at(1, 0); !t) {
824 throw;
825 }
826 return ret;
827 }
828
829 matrix<double> ret(n, n);
830 matrix<double> tmp = mt;
831
832 auto swap = [&](uint y1, uint y2) {
833 for (uint x = 0; x < n; x++) {
834 std::swap(tmp.at(y1, x), tmp.at(y2, x));
835 std::swap(ret.at(y1, x), ret.at(y2, x));
836 }
837 };
838
839 ret = identity<double>(n);
840
841 for (uint i = 0; i < n; i++) {
842 double buf = tmp.at(i, i);
843 if (buf == 0) {
844 double max = 0;
845 int max_i = i;
846 for (uint j = i + 1; j < n; j++) {
847 if (max < tmp.at(j, i)) {
848 max = tmp.at(j, i);
849 max_i = j;
850 }
851 }
